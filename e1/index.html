    public static void posiciones(String w[][]) {
        for (int i = 0; i < 5; i++) {
            for (int j = 0; j < 5; j++) { // RECORRER ARREGLO 5X5
                System.out.print(i); // IMPRIMIR NUMERO DE RENGLON
                System.out.print(j); // IMPRIMIR NUMERO DE COLUMNA
                System.out.print(" ");
            }
            System.out.println(" ");
        }
    }

    public static void inicializar(String w[][]) {
        for (int i = 0; i < 5; i++) {
            for (int j = 0; j < 5; j++) { // RECORRER ARREGLO 5X5
                w[i][j] = "#"; // LLENAR ARREGLO CON "#"
            }
        }
    }

    public static void despliega(String w[][]) {
        for (int j = 0; j < 5; j++) {
            for (int k = 0; k < 5; k++) { // RECORRER ARREGLO 5X5
                System.out.print(w[j][k]); // MOSTRAR CADA ELEMENTO DEL ARREGLO
                System.out.print(" ");
            }
            System.out.println("");
        }
        System.out.println("---------------");
    }

    public static void jugador1(String gato[][]) {
        Scanner sc = new Scanner(System.in); // INICIALIZAR SCANNER
        int x, y; // DECLARAR X Y PARA POSICIONES
        boolean entero = true, desocupado = true; // DECLARAR BOOLEANS PARA VERIFICAR LOS NUMEROS INGRESADORS
        do {
            do {
                System.out.println("\nJugador. Ingrese la posición que desee"); // IMPRIMIR QUE ES TURNO DEL JUGADOR
                System.out.print("Posición X: ");
                x = sc.nextInt(); // LEER RENGLON
                System.out.print("Posición Y: ");
                y = sc.nextInt(); // LEER COLUMNA
                if (x < 0 || y < 0 || x > 4 || y > 4) { // VERIFICAR SI ESTA FUERA DEL RANGO
                    entero = false; // RETORNAR FALSE
                    System.out.println("Posición inválida"); //MENSAJE DE ERROR
                } else { // SI ESTA DENTRO DE RANGO
                    entero = true; // RETORNAR TRUE
                }
            } while (entero == false); // REPETIR EL CICLO SI LOS NUMEROS ESTAN FUERA DE RANGO
            do {
                if (!"#".equals(gato[x][y])) { // VERIFICAR SI LA POSICION YA ESTA OCUPADA
                    System.out.println("Posición inválida"); // MENSAJE DE ERROR SI ESTA OCUPADA
                } else { // SI NO ESTA OCUPADA
                    desocupado = true; // RETORNAR TRUE
                }
            } while (desocupado == false); // REPETIR EL CICLO MIENTRAS LA POSICION ESTE OCUPADA
        } while (desocupado == false || entero == false); // REPETIR EL CICLO MIENTRAS ESTA OCUPADA O MIENTRAS LOS NUMEROS SEAN INVALIDOS
        gato[x][y] = "X"; // ASIGNAR X A LA POSICION
    }

    public static void pc(String gato[][]) {
        int x, y; // DECLARAR POSICIONES X Y
        do {
            x = (int) (Math.random() * 5); // ASIGNAR UN NUMERO ALEATORIO ENTERO 0-5 AL RENGLON
            y = (int) (Math.random() * 5); // ASIGNAR UN NUMERO ALEATORIO ENTERO 0-5 A LA COLUMNA
        } while (!"#".equals(gato[x][y])); // REPETIR EL CICLO MIENTRAS LA CASILLA ESTE OCUPADA
        gato[x][y] = "O"; // ASIGNAR O A LA CASILLA
        System.out.println("PC"); // IMPRIMIR QUE ES TURNO DE LA PC
    }
