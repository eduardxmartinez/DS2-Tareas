package ds2.tarea14;

 *

import java.util.*;

public class ejecutaAlumno {

    public static void main(String args[]) {
        Scanner sc = new Scanner(System.in); // Variable scanner
        String nombre; // Variable para guardar nombre
        String apellido; // Variable para guardar apellido
        int edad; // Variable para guardar edad
        Alumno[] alumno = new Alumno[50]; // Creamos array de 50 objetos alumno
        
        for (int i = 0; i < alumno.length; i++) { // Ciclo para añadir información a cada alumno
         nombre = sc.next(); // Ingresamos nombre
        apellido = sc.next(); // Ingresamos apellido
        edad = sc.nextInt(); // Ingresamos edad
        alumno[i] = new Alumno(nombre, apellido, edad); // Creamos objeto alumno[i]
            if (i<=9) {
             alumno[i].setExpediente("21920843"+i); // Ejecutamos método setExpediente()      
            }
            if (i>=10) {
             alumno[i].setExpediente("2192085"+i); // Ejecutamos método setExpediente()         
            }
        }
        
        for (int i = 0; i < alumno.length ; i++) { // Ciclo para mostrar información de cada alumno
         System.out.print("Alumno "+(i+1)+": "+alumno[i].getNombre() + " " + alumno[i].getApellido() + " " + alumno[i].getEdad() + " "); // Ejecutamos métodos getNombre(), getApellido() y getEdad()
        System.out.println(alumno[i].getExpediente()); // Ejecutamos método getExpediente() para cada alumno 
        }

    }
}
