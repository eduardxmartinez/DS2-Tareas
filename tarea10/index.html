package ds2.tarea10.pkg11;

public class Perro {

    // Variables de instancia
    String nombre;
    String raza;
    int edad;
    String color;
 
    // Declaración del constructor de clase
    public Perro(String nombre, String raza,
                   int edad, String color)
    {
        this.nombre = nombre;
        this.raza= raza;
        this.edad = edad;
        this.color = color;
    }
 
    public String getNombre() {
        return nombre;
    }
 
    public String getRaza() {
        return raza;
    }
 
    public int getEdad() {
        return edad;
    }
 
    public String getColor() {
        return color;
    }
    
    public void setNombre(String z) {
        this.nombre = z;
    }
    
    public void setRaza(String z) {
        this.raza = z;
    }
    
    public void putEdad(int z) {
        this.edad = z;
    }
    
   public void setColor(String z) {
        this.color = z;
    }
}
