public class DS2Tarea9 {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n;
        int[][] A = new int[5][5];
        System.out.println("Ingrese un número entero");
        n = sc.nextInt();
        for (int i = 0; i < 5; i++) {
            for (int j = 0; j < 5; j++) {
                if (i == j) {
                    A[i][j] = n;
                }
                if (i == (j - 1)) {
                    A[i][j] = n + 1;
                }
                if (i == (j + 1)) {
                    A[i][j] = n - 1;
                }
                System.out.print(A[i][j] + " ");
            }
            System.out.println("");
        }
        for (int i = 0; i < 5; i++) {
            for (int j = 0; j < 5; j++) {
            }
            System.out.println("");
        }
    }

}
