package ds2.tarea12;

import java.util.Arrays;

public class SH {

    // Variables de instancia
    String nombre;
    String raza;
    boolean capa;
    String[] superP;
 
    // Declaración del constructor de clase
    public SH(String nombre, String raza,boolean capa, String[] superP)
    {
        this.nombre = nombre;
        this.raza = raza;
        this.capa = capa;
        this.superP = superP;
    }
 
    public String getNombre() {
        return nombre;
    }
 
    public String getRaza() {
        return raza;
    }
 
    public boolean getCapa() {
        return capa ;
    }
    public String getSuperP(){
        String SP = Arrays.toString(superP);
        return SP;
    }
    
    public void setNombre(String z) {
        this.nombre = z;
    }
    
    public void setRaza(String z) {
        this.raza = z;
    }
    
   public void setCapa(boolean capa) {
        this.capa = capa;
    }
}
